+// Main Component

+export default function MessageBubble({ content, isUser }: MessageBubbleProps) {
+  // User messages: simple text
+  if (isUser) {
+    return <Text style={s.userText}>{content}</Text>;
+  }
+
+  // AI messages: parse and render markdown + math blocks
+  const blocks = useMemo(() => parseContent(content || ''), [content]);
+  const hasMath = blocks.some(b => b.type !== 'text');
+
+  // No math → pure native markdown (fastest path)
+  if (!hasMath) {
+    return (
+      <View style={s.aiContainer}>
+        <Markdown style={mdStyles}>{content || ''}</Markdown>
+      </View>
+    );
+  }
+
+  // Mixed: markdown blocks rendered natively, math blocks via KaTeX WebView
+  return (
+    <View style={s.aiContainer}>
+      {blocks.map((block, i) => {
+        if (block.type === 'text') {
+          const trimmed = block.value.trim();
+          if (!trimmed) return null;

+            math={block.value}
+            block={block.type === 'math-block'}
+          />
+        );
+      })}
+    </View>
+  );
+}
